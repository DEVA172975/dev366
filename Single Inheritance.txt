package nov19;

public class SingleInherit {
    public static void main(String[] args) {

        Son s = new Son();
        s.bike();        // child method
        s.oldBike();     // inherited from parent
    }
}

class Father {
    void oldBike() {
        System.out.println("Hero Splendor 2005");
    }
}

class Son extends Father {
    void bike() {
        System.out.println("Yamaha R15");
    }
}
